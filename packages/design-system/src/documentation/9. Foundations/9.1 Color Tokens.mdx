import { Meta } from '@storybook/addon-docs/blocks';
import tokenFile from '../../generated/uni-token.generated';
import { TokenReferenceStyles } from './tokenReferenceDocs';

<Meta title="Documentation/9. Tokens & Foundations/9.1 Color Tokens" />

export const sortByName = entries => [...entries].sort(([a], [b]) => a.localeCompare(b, undefined, { numeric: true }));
export const toPairs = source => sortByName(Object.entries(source || {}).filter(([, value]) => typeof value === 'string'));

export const semanticBrand = toPairs(tokenFile.ucolor?.brand);
export const semanticBorder = toPairs(Object.fromEntries(toPairs(tokenFile.ucolor).filter(([name]) => name.startsWith('border-'))));
export const semanticFill = toPairs(Object.fromEntries(toPairs(tokenFile.ucolor).filter(([name]) => name.startsWith('fill-'))));
export const semanticContentText = toPairs(
  Object.fromEntries(toPairs(tokenFile.ucolor?.content).filter(([name]) => name.startsWith('text-')))
);
export const semanticContentIcon = toPairs(
  Object.fromEntries(toPairs(tokenFile.ucolor?.content).filter(([name]) => name.startsWith('icon-')))
);
export const semanticGroups = [
  { id: 'border', label: 'Border', tokens: semanticBorder },
  { id: 'fill', label: 'Fill', tokens: semanticFill },
  { id: 'text', label: 'Content Text', tokens: semanticContentText },
  { id: 'icon', label: 'Content Icon', tokens: semanticContentIcon },
];

export const baseColorGroups = sortByName(
  Object.entries(tokenFile.bcolor || {}).filter(([, value]) => value && typeof value === 'object')
).map(([groupName, groupTokens]) => ({ groupName, tokens: toPairs(groupTokens) }));
