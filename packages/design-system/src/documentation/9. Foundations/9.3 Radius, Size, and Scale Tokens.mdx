export const toEntries = source => Object.entries(source || {});

export const parseTokenOrder = tokenName => {
  if (tokenName.includes('none')) return -1;
  const match = tokenName.match(/(\d+)$/);
  return match ? Number(match[1]) : Number.MAX_SAFE_INTEGER;
};

export const sortScaleEntries = entries =>
  [...entries].sort(([a], [b]) => parseTokenOrder(a) - parseTokenOrder(b) || a.localeCompare(b));

export const bsize = sortScaleEntries(toEntries(tokenFile.bsize));
export const usize = sortScaleEntries(toEntries(tokenFile.usize));
export const uspacing = sortScaleEntries(toEntries(tokenFile.uspacing));
export const bradius = sortScaleEntries(toEntries(tokenFile.bradius));
export const uradius = sortScaleEntries(toEntries(tokenFile.uradius));
export const blinewidth = sortScaleEntries(toEntries(tokenFile.blinewidth));
export const ulinewidth = sortScaleEntries(toEntries(tokenFile.ulinewidth));
export const bscreen = sortScaleEntries(toEntries(tokenFile.bscreen));
export const uscreen = sortScaleEntries(toEntries(tokenFile.uscreen));

export const renderScale = (rows, scale = 1, label = 'px') => (
  <div className="token-ref-scale-list">
    {rows.map(([tokenName, tokenValue]) => {
      const numeric = Number(tokenValue);
      const width = Math.max(2, Math.min(260, numeric / scale));
      return (
        <div key={tokenName} className="token-ref-scale-row">
          <div className="token-ref-type-token">{tokenName}</div>
          <div className="token-ref-scale-track">
            <span className="token-ref-scale-fill" style={{ width }} />
          </div>
          <div className="token-ref-type-value">{`${tokenValue}${label}`}</div>
        </div>
      );
    })}
  </div>
);
