import { Meta } from '@storybook/addon-docs/blocks';
import tokenFile from '../../generated/uni-token.generated';
import { TokenReferenceStyles } from './tokenReferenceDocs';

<Meta title="Documentation/9. Tokens & Foundations/9.2 Typography Tokens" />

export const typography = tokenFile.btypography || {};
export const fontFamily = typography['font-family']?.['font-inter'] || 'Inter';

export const numericSuffix = tokenName => {
  const match = tokenName.match(/(\d+)$/);
  return match ? Number(match[1]) : Number.MAX_SAFE_INTEGER;
};

export const sortByNumericSuffix = entries => [...entries].sort(([a], [b]) => numericSuffix(a) - numericSuffix(b));

export const fontSizes = sortByNumericSuffix(
  Object.entries(typography).filter(([name]) => name.startsWith('font-size-'))
);

export const lineHeights = sortByNumericSuffix(
  Object.entries(typography).filter(([name]) => name.startsWith('line-height-'))
);

export const normalWeights = sortByNumericSuffix(Object.entries(typography['font-weight']?.normal || {}));

export const sampleText = 'Sphinx of black quartz, judge my vow.';

<TokenReferenceStyles />

# Tokens & Foundations: Typography Tokens

<div className="token-ref-doc">
  <div className="token-ref-callout">
    Typography is defined in <code>btypography</code>. Consume semantic text styles in product UI, backed by this foundation scale.
  </div>

  <div className="token-ref-section">
    <h3>Font Family</h3>
    <p>
      Primary token: <code>font-family.font-inter</code> = <strong>{fontFamily}</strong>
    </p>
  </div>

  <div className="token-ref-section">
    <h3>Font Size Specimens</h3>
    {fontSizes.map(([tokenName, tokenValue]) => (
      <div key={tokenName} className="token-ref-type-row">
        <div className="token-ref-type-token">{tokenName}</div>
        <div style={{ fontFamily, fontSize: Number(tokenValue), lineHeight: 1.3 }}>{sampleText}</div>
        <div className="token-ref-type-value">{`${tokenValue}px`}</div>
      </div>
    ))}
  </div>

  <div className="token-ref-section">
    <h3>Line Height</h3>
    {lineHeights.map(([tokenName, tokenValue]) => (
      <div key={tokenName} className="token-ref-type-row">
        <div className="token-ref-type-token">{tokenName}</div>
        <div style={{ fontFamily, fontSize: 14, lineHeight: Number(tokenValue) / 14 }}>{sampleText}</div>
        <div className="token-ref-type-value">{`${tokenValue}px`}</div>
      </div>
    ))}
  </div>

  <div className="token-ref-section">
    <h3>Weight Scale</h3>
    {normalWeights.map(([tokenName, tokenValue]) => (
      <div key={tokenName} className="token-ref-type-row">
        <div className="token-ref-type-token">{tokenName}</div>
        <div style={{ fontFamily, fontSize: 18, fontWeight: Number(tokenValue), lineHeight: 1.3 }}>{sampleText}</div>
        <div className="token-ref-type-value">{String(tokenValue)}</div>
      </div>
    ))}
  </div>
</div>
